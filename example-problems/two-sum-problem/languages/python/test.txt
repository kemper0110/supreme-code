from solution import two_sum


CASES = [
    ([1, 2, 3], 4),
    ([1234, 5678, 9012], 14690),
    ([2, 2, 3], 4),
    ([2, 3, 1], 4),
]


def test_two_sum_cases():
    for nums, target in CASES:
        result = two_sum(nums, target)

        assert len(result) == 2
        assert result[0] != result[1]
        assert nums[result[0]] + nums[result[1]] == target
